local function normalizeName(name)
	if not name then return nil end
	name = name:gsub("%s+", " ")
	name = name:gsub("^%s+", ""):gsub("%s+$", "")
	return name
end

local function stripGameSuffixes(name)
	if not name then return nil end
	name = name:gsub("%s*%b()", "")
	name = name:gsub("^%s+", ""):gsub("%s+$", "")
	return name
end

local function canonicalGameName(name)
	return normalizeName(stripGameSuffixes(name))
end

local function lookupKey(name)
	return name:lower():gsub("_", " ")
end

local function normalizeLookupKey(name)
	if not name then return nil end
	return name:lower():gsub("[^%w]", "")
end

local AMVGG_PARTIAL_RULES = {
	{ patterns = { "fly%-a", "fly a" }, amvgg = "Fly-A-Pet Potion", value = 0.012 },
	{ patterns = { "ride%-a", "ride a" }, amvgg = "Ride-A-Pet Potion", value = 0.0065 },
}

local function matchesPartialRule(lower, patterns)
	for _, pattern in ipairs(patterns) do
		if lower:find(pattern, 1, false) then
			return true
		end
	end
	return false
end

local function getAmvggItemByName(amvggName)
	return simplebyname[amvggName] or simplebylower[amvggName:lower()]
end

local function getInventoryInfo(kind, category)
	if not kind then
		return nil
	end
	if category and InventoryDB[category] and InventoryDB[category][kind] then
		return InventoryDB[category][kind]
	end
	for _, catTable in pairs(InventoryDB) do
		if type(catTable) == "table" and catTable[kind] then
			return catTable[kind]
		end
	end
	return nil
end

local function collectItemSearchNames(kind, category, item_data)
	local names, seen = {}, {}

	local function add(name)
		if name == nil then
			return
		end
		local n = normalizeName(tostring(name))
		if n and not seen[n] then
			seen[n] = true
			table.insert(names, n)
		end
		local c = canonicalGameName(n)
		if c and not seen[c] then
			seen[c] = true
			table.insert(names, c)
		end
	end

	if item_data then
		add(item_data.kind)
		if item_data.id and KindDB[item_data.id] then
			local kd = KindDB[item_data.id]
			add(kd.name)
			add(kd.kind)
			add(kd.display_name)
		end
	end

	if kind then
		add(kind)
		add(kindnames[kind])
		local info = getInventoryInfo(kind, category)
		if info then
			add(info.name)
		end
	end

	return names
end

local function resolveKind(item_data)
	if not item_data then
		return nil
	end

	local kind = item_data.kind
	if not kind and item_data.id then
		local kd = KindDB[item_data.id]
		if kd then
			kind = kd.name or kd.kind
		end
	end

	return kind
end

local function lookupPet(petkind)
	if not petkind then return nil end

	local display = kindnames[petkind]
	if display then
		local pet = petsbyname[display] or petsbylower[display:lower()]
		if pet then return pet end
	end

	local n = normalizeName(petkind)
	if n then
		local pet = petsbyname[n] or petsbylower[n:lower()]
		if pet then return pet end
	end

	return petsbylower[lookupKey(petkind)]
end

local function findSimpleByDisplayName(name)
	if not name then
		return nil
	end

	for _, candidate in ipairs({
		normalizeName(name),
		canonicalGameName(name),
	}) do
		if candidate then
			local item = simplebyname[candidate] or simplebylower[candidate:lower()]
			if item then
				return item
			end
			local key = normalizeLookupKey(candidate)
			if key and simplebykey[key] then
				return simplebykey[key]
			end
		end
	end

	local stripped = canonicalGameName(name)
	if stripped then
		local lower = stripped:lower()
		local best, bestLen = nil, 0
		for amvggLower, item in pairs(simplebylower) do
			if lower:sub(1, #amvggLower) == amvggLower and #amvggLower > bestLen then
				best = item
				bestLen = #amvggLower
			end
		end
		if best then
			return best
		end
	end

	return nil
end

local function findSimpleByPartialName(name)
	if not name then
		return nil
	end

	local lower = name:lower()
	for _, rule in ipairs(AMVGG_PARTIAL_RULES) do
		if matchesPartialRule(lower, rule.patterns) then
			local item = getAmvggItemByName(rule.amvgg)
			if item then
				return item
			end
			if rule.value then
				return { name = rule.amvgg, value = tostring(rule.value) }
			end
		end
	end

	return findSimpleByDisplayName(name)
end

local function lookupSimple(kind, category, item_data)
	local lookupKind = (item_data and item_data.kind) or kind

	if lookupKind and simplebykind[lookupKind] then
		return simplebykind[lookupKind]
	end
	if kind and simplebykind[kind] then
		return simplebykind[kind]
	end

	for _, name in ipairs(collectItemSearchNames(kind, category, item_data)) do
		local item = findSimpleByPartialName(name)
		if item then
			return item
		end
	end

	if kind then
		local item = findSimpleByPartialName(kind)
		if item then
			return item
		end
		return simplebylower[lookupKey(kind)]
	end

	return nil
end

local function rebuildFoodPartialAliases()
	for kindId, info in pairs(InventoryDB.food or {}) do
		if type(info) == "table" and info.name then
			local lower = info.name:lower()
			for _, rule in ipairs(AMVGG_PARTIAL_RULES) do
				if matchesPartialRule(lower, rule.patterns) then
					local item = getAmvggItemByName(rule.amvgg)
					if not item and rule.value then
						item = { name = rule.amvgg, value = tostring(rule.value) }
					end
					if item then
						simplebykind[kindId] = item
					end
					break
				end
			end
		end
	end
end

local function rebuildSimpleByKind()
	for k in pairs(simplebykind) do
		simplebykind[k] = nil
	end
	for catName, catTable in pairs(InventoryDB) do
		if catName ~= "pets" and type(catTable) == "table" then
			for kindId, info in pairs(catTable) do
				if type(info) == "table" and info.name then
					local item = findSimpleByPartialName(info.name)
					if item then
						simplebykind[kindId] = item
					end
				end
			end
		end
	end
	rebuildFoodPartialAliases()
end

local function indexSimpleItem(item)
	local n = normalizeName(item.name)
	if not n then return end
	simplebyname[n] = item
	simplebylower[n:lower()] = item
	local key = normalizeLookupKey(n)
	if key then
		simplebykey[key] = item
	end
end

local function roundFixed(n, decimals)
	local mult = 10 ^ decimals
	return math.floor(n * mult + 0.5) / mult
end

local function computeDerivedValues(pet)
	local cat = tonumber(pet.category)
	if not cat or cat == 13 then
		return nil
	end

	local coeffs = CATEGORY_COEFFS[tostring(cat)]
	if not coeffs then
		return nil
	end

	local reg = tonumber(pet.regularValue) or 0
	local neon = tonumber(pet.neonValue) or 0
	local mega = tonumber(pet.megaValue) or 0
	local s0 = reg >= 0.0175 and 3 or 4
	local s1 = 4

	if cat == 11 and reg > 0.08 then
		return {
			NP = roundFixed(0.95 * reg, s0),
			R = roundFixed(0.975 * reg, s0),
			F = roundFixed(0.975 * reg, s0),
			NNP = roundFixed(neon * coeffs.NNP, s1),
			NR = neon > 0.2 and neon or roundFixed(neon * coeffs.NR, s1),
			NF = neon > 0.2 and neon or roundFixed(neon * coeffs.NF, s1),
			MNP = roundFixed(mega * coeffs.MNP, s1),
			MR = mega > 0.9 and mega or roundFixed(mega * coeffs.MR, s1),
			MF = mega > 0.9 and mega or roundFixed(mega * coeffs.MF, s1),
		}
	end

	return {
		NP = roundFixed(reg * coeffs.NP, s0),
		R = roundFixed(reg * coeffs.R, s0),
		F = roundFixed(reg * coeffs.F, s0),
		NNP = roundFixed(neon * coeffs.NNP, s1),
		NR = roundFixed(neon * coeffs.NR, s1),
		NF = roundFixed(neon * coeffs.NF, s1),
		MNP = roundFixed(mega * coeffs.MNP, s1),
		MR = roundFixed(mega * coeffs.MR, s1),
		MF = roundFixed(mega * coeffs.MF, s1),
	}
end

local function getVariantKey(props)
	local fly = props.flyable == true
	local ride = props.rideable == true
	local pot = (fly and "f" or "") .. (ride and "r" or "")

	if props.mega_neon == true then
		return "m" .. pot
	elseif props.neon == true then
		return "n" .. pot
	end
	return pot
end

local function getBaseValue(pet, variantKey)
	local cat = tonumber(pet.category)

	if cat == 13 then
		local field = CAT13_FIELDS[variantKey]
		if field then
			return tonumber(pet[field])
		end
		return nil
	end

	local derived = computeDerivedValues(pet)
	if not derived then
		if variantKey == "fr" or variantKey == "nfr" or variantKey == "mfr" then
			if variantKey == "nfr" then return tonumber(pet.neonValue) end
			if variantKey == "mfr" then return tonumber(pet.megaValue) end
			return tonumber(pet.regularValue)
		end
		return nil
	end

	local slot = DERIVED_FIELDS[variantKey]
	if slot == "regular" then return tonumber(pet.regularValue) end
	if slot == "neon" then return tonumber(pet.neonValue) end
	if slot == "mega" then return tonumber(pet.megaValue) end
	return derived[slot]
end

local function frostDecimals(x)
	local a = math.abs(x)
	local decimals
	if a >= 100 then
		decimals = 0
	elseif a >= 10 then
		decimals = 1
	elseif a >= 1 then
		decimals = 2
	elseif a >= 0.1 then
		decimals = 3
	else
		decimals = 4
	end
	local mult = 10 ^ decimals
	return math.floor(x * mult + 0.5) / mult
end

local function toFrostValue(v)
	return frostDecimals(FROST_RATIO * v)
end

local function toRidePotValue(v)
	local stepped = math.floor(RIDEPOT_MULT * v * 20 + 0.5) / 20
	return math.floor(stepped * 100 + 0.5) / 100
end

local function extractJsonArray(body, arrayKey)
	body = body:gsub('\\u0026', '&')
	body = body:gsub('\\"', '"')

	local marker = '"' .. arrayKey .. '":'
	local startIdx = body:find(marker .. "%[")
	if not startIdx then
		return nil
	end

	startIdx = startIdx + #marker
	local depth = 0
	local endIdx = startIdx
	local inString = false
	local escape = false

	for i = startIdx, math.min(startIdx + 4000000, #body) do
		local c = body:sub(i, i)

		if inString then
			if escape then
				escape = false
			elseif c == '\\' then
				escape = true
			elseif c == '"' then
				inString = false
			end
		else
			if c == '"' then
				inString = true
			elseif c == '[' then
				depth = depth + 1
			elseif c == ']' then
				depth = depth - 1
			end
		end

		if depth == 0 and i > startIdx then
			endIdx = i
			break
		end
	end

	local json = body:sub(startIdx, endIdx)
	local ok, data = pcall(function()
		return HttpService:JSONDecode(json)
	end)
	if ok and data then
		return data
	end
	return nil
end

local function fetchAmvggList(url, arrayKey)
	local ok, response = pcall(function()
		return request({
			Url = url,
			Method = "GET",
			Headers = {
				["Accept"] = "text/html",
				["User-Agent"] = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/145.0.0.0 Safari/537.36",
			},
		})
	end)

	if not ok or not response or not response.Success then
		return nil
	end

	return extractJsonArray(response.Body, arrayKey)
end

local function rawVal(kind, props, category, item_data)
	if category and category ~= "pets" then
		local simple = lookupSimple(kind, category, item_data)
		if simple then
			return tonumber(simple.value) or 0
		end
	end

	local pet = lookupPet(kind)
	if pet then
		props = props or {}
		local key = getVariantKey(props)
		return getBaseValue(pet, key) or 0
	end

	local simple = lookupSimple(kind, category, item_data)
	if simple then
		return tonumber(simple.value) or 0
	end

	return 0
end

local function val(kind, props, category, item_data)
	local v = rawVal(kind, props, category, item_data)

	if Values.MODE == "ridepot" then
		return toRidePotValue(v)
	elseif Values.MODE == "frost" then
		return toFrostValue(v)
	end

	return v
end

local function safeGetOffer(getter)
	local ok, offer = pcall(getter)
	if not ok or type(offer) ~= "table" or type(offer.items) ~= "table" then
		return nil
	end
	return offer
end

local function GetItemsFromOffer(offer)
	if not offer or type(offer.items) ~= "table" then
		return {}, 0
	end

	local list = offer.items
	local items, total = {}, 0
	for _, item in ipairs(list) do
		local category = item.category
		local kind = resolveKind(item)
		local value = val(kind, item.properties, category, item)
		total += value
		table.insert(items, {
			name = kind or (item.kind or "?"),
			value = value,
			props = item.properties,
			category = category,
		})
	end
	return items, total
end

local function GetItemsFromMyOffer()
	local offer = safeGetOffer(function()
		return TradeApp:_get_my_offer()
	end)
	return GetItemsFromOffer(offer)
end

local function GetItemsFromPartnerOffer()
	local offer = safeGetOffer(function()
		return TradeApp:_get_partner_offer()
	end)
	return GetItemsFromOffer(offer)
end
local function FormatValue(v)
	if v == 0 then return "0" end
	if v >= 1000 then return string.format("%.1fk", v/1000) end
	if v >= 100 then return string.format("%.0f", v) end
	local s = string.format("%.4f", v)
	s = s:gsub("%.?0+$", "")
	return s
end
local function getSingleVal(itemName, props, category)
	return val(itemName, props or { flyable = true, rideable = true }, category or "pets")
end

	end
	if not valid then
		warn("[amvgg] Unknown VALUE_MODE: " .. tostring(mode))
		return false
	end
	Values.MODE = mode
	return true
end

function CrashBar.Update(myTotal, partnerTotal)
	myTotal     = myTotal or 0
	partnerTotal = partnerTotal or 0

	leftValue.Text  = string.format("%.2f", myTotal)
	rightValue.Text = string.format("%.2f", partnerTotal)

	local diff = myTotal - partnerTotal
	local sign = diff >= 0 and "+" or ""
	diffLabel.Text = sign .. string.format("%.2f", diff)

	local sum = myTotal + partnerTotal
	local ratio = sum > 0 and (myTotal / sum) or 0.5

	CrashBar.SetCategory(ratio)
	CrashBar.SetFillRatio(ratio)
end

local function attachValueLabel(frame, data)
	if typeof(frame) ~= "Instance" or not data then
		return
	end

	local item_data = data.item_data
	if not item_data and data.kind then
		item_data = data
	end
	if not item_data then
		return
	end

	local category = item_data.category or data.category
	if not category then
		return
	end

	local parent = findLabelParent(frame)
	if category == "food" then
		parent = frame
	end
	if not parent then
		return
	end

	local kind = resolveKind(item_data)
	local v = 0
	local valOk, valResult = pcall(function()
		return Values.val(kind, item_data.properties, category, item_data)
	end)
	if valOk and type(valResult) == "number" then
		v = valResult
	end

	local existing = parent:FindFirstChild("AmvggValue", true)
	if existing then
		existing:Destroy()
	end

	local label = Instance.new("TextLabel")
	label.Name = "AmvggValue"
	label.Size = UDim2.new(1.5, 0, 0, 20)
	label.Position = category == "food"

	label.Font = Enum.Font.GothamBold
	label.TextXAlignment = Enum.TextXAlignment.Center
	label.TextYAlignment = Enum.TextYAlignment.Top
	label.ZIndex = math.max(parent.ZIndex + 10, 25)
	label.ClipsDescendants = false
	label.Text = FormatValue(v)
	label.Parent = parent
end

local CREATOR_RENDER_METHODS = { "item", "pet" }
local hookedCreatorFns = {}

local function hookCreatorMethod(creator, methodName)
	local original = creator[methodName]
	if type(original) ~= "function" then
		return
	end

	local wrapped = hookedCreatorFns[original]
	if not wrapped then
		wrapped = function(self, data, ...)
			local results = table.pack(original(self, data, ...))
			if data and (data.item_data or data.kind) and typeof(results[1]) == "Instance" then
				pcall(attachValueLabel, results[1], data)
			end
			return table.unpack(results, 1, results.n)
		end
		hookedCreatorFns[original] = wrapped
	end

	creator[methodName] = wrapped
end

local function hookBackpackCreators()
	local seen = {}
	local creators = { BackpackButtonCreator }
	if BackpackApp and BackpackApp.button_creator then
		table.insert(creators, BackpackApp.button_creator)
	end

	for _, creator in ipairs(creators) do
		if creator and not seen[creator] then
			seen[creator] = true
			for _, methodName in ipairs(CREATOR_RENDER_METHODS) do
				hookCreatorMethod(creator, methodName)
			end
		end
	end
end

local function findButtonFrame(btnData)
	if not btnData then
		return nil
	end

	if typeof(btnData.frame) == "Instance" then
		return btnData.frame
	end
	if typeof(btnData.button) == "Instance" then
		return btnData.button
	end

	local unique = btnData.unique or (btnData.item_data and btnData.item_data.unique)
	local hash = btnData.hash
	if not BackpackApp then
		return nil
	end

	local roots = { BackpackApp.scroll, BackpackApp.content }
	for _, root in ipairs(roots) do
		if root then
			for _, key in ipairs({ unique, hash, unique and tostring(unique), hash and tostring(hash) }) do
				if key then
					local found = root:FindFirstChild(key, true)
					if found then
						return found
					end
				end
			end
		end
	end

	return nil
end

local function getMyInventoryItems()
	local items = {}
	if not BackpackApp or type(BackpackApp.button_by_unique) ~= "table" then
		return items
	end
	for uniqueKey, btnData in pairs(BackpackApp.button_by_unique) do
		local item_data = btnData.item_data or (btnData.kind and btnData) or nil
		if item_data then
			local category = item_data.category
			if category then
				local kind = resolveKind(item_data)
				local v = 0
				pcall(function()
					v = Values.val(kind, item_data.properties, category, item_data)
				end)
				if v > 0 then
					local uid = item_data.unique or uniqueKey
					table.insert(items, {
						unique   = uid,
						kind     = kind,
						value    = v,
						category = category,
					})
				end
			end
		end
	end
	return items
end

local function getMyOfferUniqueSet()
	local set = {}
	local ok, offer = pcall(function() return TradeApp:_get_my_offer() end)
	if ok and offer and type(offer.items) == "table" then
		for _, it in ipairs(offer.items) do
			if it.unique then set[it.unique] = true end
		end
	end
	return set
end

-- Жадный подбор предметов к целевой сумме
local function greedySelect(items, target, overshootPct)
	overshootPct = overshootPct or 0.14
	table.sort(items, function(a, b) return a.value > b.value end)
	local selected, total = {}, 0
	for _, item in ipairs(items) do
		if #selected >= 18 then break end
		if total >= target then break end
		if (total + item.value) <= target * (1 + overshootPct) then
			table.insert(selected, item)
			total = total + item.value
		end
	end
	return selected, total
end

local autoFillBusy = false

function _G.doAutoFill(mode)
	if autoFillBusy then
		setSideStatus("Подождите, идёт заполнение...", Color3.fromRGB(200, 150, 30))
		return
	end
	autoFillBusy = true

	task.spawn(function()
		-- Принудительный refresh перед расчётом
		pcall(function() TradeApp:refresh_all() end)
		task.wait(0.25)

		local _, partnerTotal = Values.GetItemsFromPartnerOffer()
		partnerTotal = partnerTotal or 0

		if partnerTotal <= 0 then
			setSideStatus("Партнёр не добавил предметы!", Color3.fromRGB(200, 60, 60))
			autoFillBusy = false
			return
		end

		local cfg = FILL_TARGETS[mode]
		local targetValue = partnerTotal * cfg.ratio

		setSideStatus(
			string.format("Партнёр: %.2f → Цель: %.2f [%s]", partnerTotal, targetValue, cfg.label),
			Color3.fromRGB(50, 50, 50)
		)

		-- Снимаем текущие предметы через официальный метод TradeApp
		-- (он обновляет локальный стейт и не вызывает ресинк-баги)
		local ok2, myOffer = pcall(function() return TradeApp:_get_my_offer() end)
		if ok2 and myOffer and type(myOffer.items) == "table" and #myOffer.items > 0 then
			local toRemove = {}
			for _, it in ipairs(myOffer.items) do
				table.insert(toRemove, it)
			end
			for _, it in ipairs(toRemove) do
				pcall(function()
					TradeApp:_remove_item_from_my_offer(it)
				end)
				task.wait(0.12)
			end
			task.wait(0.25)
		end

		-- Берём предметы из инвентаря (ВСЕ категории)
		local allItems = getMyInventoryItems()
		local inOffer  = getMyOfferUniqueSet()
		local available = {}
		for _, item in ipairs(allItems) do
			if not inOffer[item.unique] then
				table.insert(available, item)
			end
		end

		if #available == 0 then
			setSideStatus("Нет предметов с известной ценностью!", Color3.fromRGB(200, 60, 60))
			autoFillBusy = false
			return
		end

		local selected, myTotal = greedySelect(available, targetValue)

		if #selected == 0 then
			setSideStatus(string.format("Нет подходящих под %.2f", targetValue), Color3.fromRGB(200, 60, 60))
			autoFillBusy = false
			return
		end

		-- Добавляем предметы с паузами (чтобы сервер успевал обрабатывать)
		local addedCount = 0
		for _, item in ipairs(selected) do
			local ok3 = pcall(function()
				RouterClient.get("TradeAPI/AddItemToOffer"):FireServer(item.unique)
			end)
			if ok3 then addedCount += 1 end
			task.wait(0.12)
		end

		-- Финальный refresh
		task.wait(0.3)
		pcall(function() TradeApp:refresh_all() end)
		task.wait(0.15)
		CrashBar.Refresh()

		setSideStatus(
			string.format("✓ [%s] %d пред. | мой: %.2f / партнёр: %.2f",
				cfg.label, addedCount, myTotal, partnerTotal),
			Color3.fromRGB(30, 140, 60)
		)
		task.delay(5, function()
			setSideStatus("Нажми кнопку для авто-заполнения")
		end)

		autoFillBusy = false
	end)
end
